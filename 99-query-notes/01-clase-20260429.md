# Revisar la calidad de esas mediciones

SELECT 
    sps.speciescode,
    sps.maxhfl,
    occ.species, 
    occ.hfl 
FROM 
    Surveys AS occ, 
    Species AS sps 
WHERE 
    sps.speciescode=occ.species AND 
    occ.hfl IS NOT NULL AND 
    occ.hfl > sps.maxhfl;
